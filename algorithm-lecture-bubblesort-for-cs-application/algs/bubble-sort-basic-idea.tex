% basic idea of bubble sort

\begin{algorithm}[H]
  \begin{algorithmic}[1]
	\Procedure{BubbleSortOverview}{$A: a_1\; a_2\; \cdots\; a_n$}
	  \Repeat
		\State Pick any $i$
		  \If{$a_{i} > a_{i+1}$}	\Comment{CAS}
			\State \Call{Swap}{$a_{i}, a_{i+1}$}
		  \EndIf
	  \Until{no adjacent inversions}
    \EndProcedure
  \end{algorithmic}
\end{algorithm}
