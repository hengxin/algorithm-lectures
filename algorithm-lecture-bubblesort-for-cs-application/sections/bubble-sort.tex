  \begin{enumerate}[1)]
	\item \textcolor{blue}{$\exists$ loop : no swaps} $\implies$ swapped = false $\implies$ terminates
	  \pause
	\item \textcolor{red}{$\forall$ loop : has swaps} \pause \qquad \textcolor{red}{\bf \textsc{Impossible}!}
  \end{enumerate}
